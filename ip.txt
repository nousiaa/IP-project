CREATE TABLE data(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id int,
    drawing_id int,
    command text,
    deleted int
);

CREATE TABLE drawing(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name text,
    owner_id int,
    description text, 
    allowed_users text, 
    deleted int
);
CREATE TABLE user(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username text NOT NULL,
    password text NOT NULL, 
    deleted int
);
